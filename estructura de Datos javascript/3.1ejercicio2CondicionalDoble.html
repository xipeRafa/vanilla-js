Algoritmo que valide si una persona es mayor o menor de edad, tenga en cuenta que una persona es mayor a partir de los 18 años

function hacer(){
				
let nombre = document.getElementById('nombre').value
	edad = document.getElementById('edad').value
	info= document.getElementById('info')
			
if(edad>=18){
	 info.innerHTML=(nombre+" es Mayor de edad por que tiene:"+edad+"años de edad")
	}
	else{
     info.innerHTML=(nombre+" es Menor de edad por que tiene:"+edad+"años de edad")
	}
				
}
