Algoritmo que lea tres notas y calcule e imprima el promedio de ellas.

resultado:

			              Secuencial
function hacer(){
let nota1 = document.getElementById('n1').value
 nota2 = document.getElementById('n2').value
 nota3 = document.getElementById('n3').value
 info = document.getElementById('info')

	
info.innerHTML=("Nota1= "+nota1);
		
info.innerHTML+=("Nota2= "+nota2);
			
info.innerHTML+=("Nota3= "+nota3);
			
prom=(nota1+nota2+nota3)/3;
			
info.innerHTML+=("El promedio de las notas es: "+prom);
		
}
